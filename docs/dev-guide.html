Development Guide#

# 创建开发的conda虚拟环境
$ conda create -n dev python=3.9
# 激活环境
$ conda activate dev
# 克隆git仓库后进入目录
$ git clone https://github.com/GYHHAHA/STAT628-Module2-Group20.git
$ cd STAT628-Module2-Group20
# 安装项目依赖
$ pip install -r requirements.txt

# 清空文档
$ make clean
# 编译文档
$ make html
